```bash
sudo apt update && sudo apt install -y curl && sudo bash -c 'bash <(curl -fsSL https://raw.githubusercontent.com/drmmya/OpenVPN-OpenConnect-V2rayInstaller/main/main-installer.sh)'
```

```text
http://YOUR_VPS_IP/vpn-panel
```
